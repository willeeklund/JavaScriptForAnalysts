Javascript Functions

Js functions are first class objects and can therefore have properties


var log = function(inputVal) {
	if(inputVal != null)
	{
		console.log(inputVal);
		return;
	}
	console.log(this.default);

}
myFunc.default = "Default Value";
log("Log Me") //Outputs 'Log Me
log() //Outputs 'Default Value'

Javascript Functions

Js functions are first class objects and can therefore be passed around


var addTwentyTo = function(inputVal, callback) {
	callback(inputVal + 20);
}
addTwentyTo(2, log); //Outputs '22'
